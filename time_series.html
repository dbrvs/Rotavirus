from numpy.fft import fft
import math
fourier = fft(proportion_cases/max(proportion_cases))

plt.figure(figsize=(12,4),dpi=600)

magz=[]
phasez=[]
for i in range(12):
    magz.append(np.sqrt(fourier[i].imag**2+fourier[i].real**2))
    phasez.append(math.atan(fourier[i].imag/fourier[i].real))


new_series=np.zeros(12)
num_ffts=6
for i in range(1,num_ffts):
    new_series += magz[i]*np.cos(i*2*np.pi*months/12 + phasez[i])
    plt.subplot(100+num_ffts*10+i)
    plt.plot(months,new_series/max(new_series))
    plt.plot(months,proportion_cases/max(proportion_cases))
